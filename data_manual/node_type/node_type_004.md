[Show: "$\mathop{\lim }\limits_{{n \rightarrow  \infty }}\sqrt[n]{a} = 1$"]

[Assume: "$a=1$"]
[Have: "$\mathop{\lim }\limits_{{n \rightarrow  \infty }}\sqrt[n]{a} = 1$"]
[Assume: "$a>1$"]
