[Show: "$(R^{-1})^{-1} = R$"]
[Show: "$(R^{-1})^{-1} \\subseteq R$"]
[Fix: {x, y} st "$(x, y) \in (R^{-1})^{-1}$"]
[Have: "$(y, x) \in R^{-1}$" by "definition of inverse"]
[Have: "$(x, y) \in R$"]
[Have: "$(R^{-1})^{-1} \subseteq R$"]
[Show: "R \subseteq (R^{-1})^{-1}"]
[Fix: {x, y} st "$(x, y) \in R$"]
[Have: "$(y, x) \in R^{-1}$"]
[Have: "$(x, y) \in (R^{-1})^{-1}$"]
[Have: "R \subseteq (R^{-1})^{-1}"]
[Have: "$(R^{-1})^{-1} = R$"]
